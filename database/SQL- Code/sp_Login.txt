CREATE PROCEDURE dbo.sp_Login

@User_Name AS NVARCHAR(500),
@Password AS NVARCHAR(500) 
AS 
BEGIN

SET NOCOUNT ON;

	SELECT  COUNT(*)
	FROM HashedCredentials
	WHERE Username = @User_Name
	AND Password = @Password
	
END;
GO
